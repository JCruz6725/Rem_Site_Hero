Table Account {
  //id int [pk, increment]
  id int //DJANGO UUID
  full_name varchar
  email email
  location varchar
  username varchar
  
  is_staff boolean
  is_active boolean
  is_admin boolean
  is_superuser boolean
  
  last_login datetime
  date_join datetime
}


Table Resume {
  user_id int [ref: > Account.id] //DJANGO UUID
  id int //DJANGO UUID
  
  title varchar
  summary varchar
   
}

Table Project{
  id int 
  user_id int [ref:  > Account.id] //DJANGO UUID
  resume_id int [ref: > Resume.id] //DJANGO UUID
  
  project_name varchar //[ref: > Resume.project_xp_name]
  //title_on_project varchar
  tech_used varchar
  summary varchar
  external_link varchar
}

Table Education {
  user_id int [ref:  > Account.id] //DJANGO UUID
  resume_id int [ref: > Resume.id] //DJANGO UUID
  id int //DJANGO UUID 
  institution_name varchar
  degree varchar
  time_at varchar
  location varchar
}


Table Professional_xp {
  id int
  user_id int [ref:  > Account.id]
  resume_id int  [ref: > Resume.id]
  employer_name varchar //[ref: > Resume.professional_xp_name]
  position varchar
  time_at varchar
  title_on_project varchar
  tech_used varchar
  summary varchar
}
